336 //BPM変化情報を基にbeat -> sec
337 [BurstCompile]
338 public float ToSec(float nowBeat,BlobAssetReference<BPMChangesBlob> bpmChanges)
339 {
340 //累計秒数
341 float accumulatedSec = 0f;
342 //BPM変化index
343 int index = 0;
344 //既に変化済みの回数
345 int alreadyChangedCount = 0;
346 //変化済み回数カウント
347 for (int i = 0; i < bpmChanges.Value.bpmChanges.Length; i++)
348 {
349 if (bpmChanges.Value.bpmChanges[i].executeBeat <= nowBeat)
350 {
351 alreadyChangedCount++;
352 }
353 else
354 {
355 break;
356 }
357 }
358
359 //変換するbeatの直前にあるBPM変化までのsecを求める
360 while(index < alreadyChangedCount - 1)
361 {
362 //変化済みの回数分秒数を足す
363 accumulatedSec +=
364 ToSecWithFixedBPM(
365 bpmChanges.Value.bpmChanges[index + 1].executeBeat - bpmChanges.Value.bpmChanges[index].executeBeat,
366 bpmChanges.Value.bpmChanges[index].afterBPM
367 );
368 index++;
369 }
370
371 //残りのbeat分を足す
372 accumulatedSec +=
373 ToSecWithFixedBPM(
374 nowBeat - bpmChanges.Value.bpmChanges[index].executeBeat,
375 bpmChanges.Value.bpmChanges[index].afterBPM
376 );
377
378 return accumulatedSec;
379 }

382 //BPM変化情報を基にsec -> beat
383 private float ToBeat(float nowSec,BlobAssetReference<BPMChangesBlob> bpmChanges)
384 {
385 //累計秒数
386 float accumulatedSec = 0f;
387 //BPM変化index
388 int index = 0;
389 //既に変化済みの回数
390 int alreadyChangedCount = bpmChanges.Value.bpmChanges.Length;
391
392 //最後から1つ前のBPM変化までループ
393 while(index < alreadyChangedCount - 1)
394 {
395 //index回目のBPM変化地点での秒数
396 float tmpSec = accumulatedSec;
397
398 //次(index + 1)のテンポ変化のタイミング(秒)の計算
399 accumulatedSec +=
400 ToSecWithFixedBPM(
401 bpmChanges.Value.bpmChanges[index + 1].executeBeat - bpmChanges.Value.bpmChanges[index].executeBeat,
402 bpmChanges.Value.bpmChanges[index].afterBPM
403 );
404
405 if(accumulatedSec >= nowSec)
406 {
407 //次のBPM変化タイミングが変換するsecを超えた場合、「超える直前のBPM変化があるbeat + 残りのbeat」を返す
408 return bpmChanges.Value.bpmChanges[index].executeBeat +
409 ToBeatWithFixedBPM(
410 nowSec - tmpSec,
411 bpmChanges.Value.bpmChanges[index].afterBPM
412 );
413 }
414
415 index++;
416
417 }
418 //変換するsecが最後のBPM変化よりも後にある場合、「最後のBPM変化があるbeat + 残りのbeat」を返す
419 return bpmChanges.Value.bpmChanges[alreadyChangedCount - 1].executeBeat +
420 ToBeatWithFixedBPM(
421 nowSec - accumulatedSec,
422 bpmChanges.Value.bpmChanges[alreadyChangedCount - 1].afterBPM
423 );
424 }
425 }
