174 private void LoadHeaderLine(string line)
175 {
176 //各ヘッダー名に対して実行
177 foreach (var headerPattern in headerPatterns)
178 {
179 Match match = Regex.Match(line, headerPattern);
180 //ヘッダー行のパターンに一致すればデータ取得
181 if (match.Success)

228 private void LoadMainLaneData(string line, float nowBeat)
229 {
230 //この小節の数字の数

252 laneCount = 0;

330 zeroCount += zero;

332 }
333 }

335
336 //1つの"0"のbeat
337 float oneBeat = 4f / zeroCount;
338
339 //BPM変化の場合
340 string[] splitB = line.Split('B');
341 //一時保持
342 float changeBPM = 0;
343 //0のカウント
344 zeroCount = 0;
345 for (var i = 0; i < splitB.Length; i++)
346 {
347 if (i % 2 == 1)

351

357 //次の0のときの0の数
