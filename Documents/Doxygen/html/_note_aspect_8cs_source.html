263 public bool WithinJudgeRangeCheck()
264 {
265 //処理済みは常にfalseを返す
266 if (noteDataRW.ValueRO.isExecuted) return false;
267
268 var type = noteDataRW.ValueRO.noteType;
269 float nowSec = noteDataRW.ValueRO.nowSec - GetJudgeOffsetSec() + GetPlayerInputOffsetSec();
270 float beginSec = noteDataRW.ValueRO.timingData.beginSec;
271 float endSec = noteDataRW.ValueRO.timingData.endSec;
272 float judgeSec = (beginSec - nowSec);
273 float missRange = GameSettings.missRange;
274
275 //判定
276 //シングル
277 if (type == NoteComponentData.NoteType.Single) return Mathf.Abs(judgeSec) <= missRange;
278 else
279 {
280 //ロング
281 bool isProcessing = noteDataRW.ValueRO.isProcessing;
282 //終点
283 if (isProcessing)
284 {
285 judgeSec = (endSec - nowSec);
286 //負なら範囲内ならtrue
287 //正なら常にtrue
288 //ロングノーツなので離れる時にしか終点判定はないはず
289 if(judgeSec < 0) return -missRange <= judgeSec;
290 return true;
291 }
292 //始点
293 return Mathf.Abs(judgeSec) <= missRange;
294 }
295 }

302 public bool ThroughNote(bool isAuto)
303 {
304 //処理済みならスキップ
305 if (noteDataRW.ValueRO.isExecuted) return false;
306
307 float nowSec = noteDataRW.ValueRO.nowSec - GetJudgeOffsetSec() + GetPlayerInputOffsetSec();
308 float judgeSec = noteDataRW.ValueRO.timingData.beginSec;
309 //処理中なら終わりのsecを入れる
310 if (noteDataRW.ValueRO.isProcessing) judgeSec = noteDataRW.ValueRO.timingData.endSec;
311
312 //判定に使う差分
313 //オートなら0
314 float judgeSecDiff = GameSettings.missRange;
315 if (isAuto) judgeSecDiff = 0;
316
317 //処理判定
318 if (nowSec - judgeSec > judgeSecDiff)
319 {
320 //オートならパーフェクトで処理
321 //オートじゃなければミスで処理
322 GameSettings.JudgeType judgeType = GameSettings.JudgeType.Miss;
323 if (isAuto) judgeType = GameSettings.JudgeType.Perfect;
324 ExecuteNote(judgeType);
325 return true;
326 }
327
328 //処理しなかった
329 return false;
330 }
