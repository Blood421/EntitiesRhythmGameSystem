386 [BurstCompile]
387 private void Execute(NotePresenterAspect notePresenterAspect)
388 {
389 //曲IDが違ったらスキップ
390 if (notePresenterAspect.GetMusicID() != judgedInputData[0].musicID) return;
391
392 float nowSec = notePresenterAspect.GetNowSec();
393 float judgeNowSec = 0;
394 float missRange = GameSettings.missRange;
395 float goodRange = GameSettings.goodRange;
396 float perfectRange = GameSettings.perfectRange;
397 float judgeSec = 0;
398 float secDif = 0;
399
400 for (int lane = 0; lane < judgeDecideNoteData.Length; lane++)
401 {
402 //無効データなのでスキップ
403 if(judgeDecideNoteData[lane].musicID != notePresenterAspect.GetMusicID()) continue;
404 //そもそも判定しないレーンはスキップ
405 if(!judgedInputData[lane].isJudge) continue;
406
407 //押されたとき
408 if (judgedInputData[lane].isPressed)
409 {
410 //この場合は例外なのでスキップ
411 if(judgeDecideNoteData[lane].isProcessing) continue;
412
413 judgeSec = judgeDecideNoteData[lane].judgeSec;
414 judgeNowSec = nowSec - judgeDecideNoteData[lane].offsetSec;
415
416 secDif = Mathf.Abs(judgeSec - judgeNowSec);
417 //パーフェクトの範囲内
418 if(secDif < perfectRange)
419 {
420 judgeResultData[lane] = new NoteJudgeSystem.JudgeResultData()
421 {
422 musicID = judgeDecideNoteData[lane].musicID,
423 entityIndex = judgeDecideNoteData[lane].entityIndex,
424 judgeType = GameSettings.JudgeType.Perfect
425 };
426 continue;
427 }
428 //Goodの範囲内
429 if(secDif < goodRange)
430 {
431 judgeResultData[lane] = new NoteJudgeSystem.JudgeResultData()
432 {
433 musicID = judgeDecideNoteData[lane].musicID,
434 entityIndex = judgeDecideNoteData[lane].entityIndex,
435 judgeType = GameSettings.JudgeType.Good
436 };
437 continue;
438 }
439
440 //上記範囲内でなければミス
441 judgeResultData[lane] = new NoteJudgeSystem.JudgeResultData()
442 {
443 musicID = judgeDecideNoteData[lane].musicID,
444 entityIndex = judgeDecideNoteData[lane].entityIndex,
445 judgeType = GameSettings.JudgeType.Miss
446 };
447
448 continue;
449 }
450
451 //離れたとき
452 //離れたときに処理中でないノーツはスキップ
453 if(!judgeDecideNoteData[lane].isProcessing) continue;
454 judgeSec = judgeDecideNoteData[lane].judgeSec;
455 judgeNowSec = nowSec - judgeDecideNoteData[lane].offsetSec;
456
457 secDif = Mathf.Abs(judgeSec - judgeNowSec);
458 //パーフェクトの範囲内
459 if(secDif < perfectRange)
460 {
461 judgeResultData[lane] = new NoteJudgeSystem.JudgeResultData()
462 {
463 musicID = judgeDecideNoteData[lane].musicID,
464 entityIndex = judgeDecideNoteData[lane].entityIndex,
465 judgeType = GameSettings.JudgeType.Perfect
466 };

468 }
